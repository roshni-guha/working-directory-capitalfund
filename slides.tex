  \begin{columns}[T]
    \begin{column}{0.50\textwidth}
      \large\textbf{The idea in one sentence}\\[0.6em]
      \normalsize
      Every day, the model looks at the global macro environment and picks \alert{one market} to hold --- rotating dynamically as conditions change.\\[1em]

      \textbf{The 6 markets it can choose:}
      \vspace{0.3em}
      \begin{itemize}\normalsize
        \setlength\itemsep{0.3em}
        \item US equities (SPY)
        \item Japan (EWJ)
        \item India (INDA)
        \item China (MCHI)
        \item Europe (EZU)
        \item Cash (BIL)
      \end{itemize}
    \end{column}
    \begin{column}{0.46\textwidth}
      \begin{block}{What it reads to decide}
        \vspace{0.3em}\normalsize
        \begin{itemize}
          \setlength\itemsep{0.3em}
          \item Fed balance sheet (QE / tightening)
          \item US M2 money supply
          \item Long bond prices (TLT)
          \item VIX / market fear (VIXY)
          \item Yield curve slope (10Y$-$2Y)
          \item S\&P 500 earnings and growth
          \item Momentum signals (BMOM)
        \end{itemize}
        \vspace{0.3em}
      \end{block}
      \vspace{0.5em}

\begin{frame}{Why These Macro Signals and BMOM}
  \begin{columns}[T]
    \begin{column}{0.50\textwidth}
      \textbf{Macro signals --- they \emph{lead} prices}\\[0.4em]
      \begin{itemize}\small
        \setlength\itemsep{0.35em}
        \item \textbf{Fed balance sheet \& M2:} liquidity injections drive risk appetite before equities move
        \item \textbf{TLT price:} falling bond prices signal rising rates --- rotates capital across regions differently
        \item \textbf{VIXY:} fear gauge; high VIX $\Rightarrow$ rotate to cash or defensive markets before momentum confirms
        \item \textbf{Yield curve (10Y$-$2Y):} inversion historically precedes recessions; slope predicts EM vs.\ DM divergence
        \item \textbf{S\&P earnings \& growth:} fundamental anchor for US-vs-EM overweight decision
      \end{itemize}
    \end{column}
    \begin{column}{0.46\textwidth}
      \textbf{BMOM --- what the macro misses}\\[0.4em]
      \normalsize
      Pure momentum captures trend persistence the macro cannot explain.\\[0.5em]
      \small
      \textbf{Formula:}
      \[
        \text{BMOM} = 0.15\,m_{20} + 0.35\,m_{60} + 0.35\,m_{120} + 0.15\,m_{240}
      \]
      Triangular weights emphasise the 3--6 month horizon --- the strongest empirically documented momentum effect.\\[0.5em]
      One BMOM score is computed for each \textbf{signal ticker} (TLT, VIXY, IBIT), giving the model 3 momentum inputs that capture trend strength in bonds, volatility, and crypto --- regimes the macro level alone misses.\\[0.5em]
      Together: macro leads, BMOM confirms.
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Policy Network and Value Network --- What They Do}
  \vspace{0.5em}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \begin{block}{Policy Network}
        \vspace{0.4em}\normalsize
        \textbf{Makes the trading decision.}\\[0.5em]
        Looks at today's macro state and outputs a probability for each of the 6 ETFs. The one with the highest probability is the trade.\\[0.5em]
        During training it gradually learns which macro conditions favour which market. During live use it just runs a forward pass and picks the top ETF.
        \vspace{0.4em}
      \end{block}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{block}{Value Network}
        \vspace{0.4em}\normalsize
        \textbf{Judges how good the situation is.}\\[0.5em]
        Given the same macro state, it outputs a single number estimating how much total reward the agent can expect from here.\\[0.5em]
        PPO uses this to compute whether a trade turned out \textit{better or worse than expected} --- that gap (the advantage) is what drives the policy update.
        \vspace{0.4em}
      \end{block}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Reward Function}
  \vspace{0.3em}
  \begin{center}
    \large\textbf{After each trade, the agent gets a score:}\\[1em]
    \begin{beamercolorbox}[wd=0.85\textwidth,center,sep=0.5em]{block title}
      \LARGE $\text{reward} \;=\; \underbrace{R_t}_{\text{today's return}} \;-\; 2.0 \times \underbrace{\text{variance of chosen ETF}}_{\text{risk penalty}}$
    \end{beamercolorbox}
  \end{center}
  \vspace{1em}
  \begin{columns}[T]
    \begin{column}{0.32\textwidth}
      \begin{block}{\centering Pick a high return}
        \centering\vspace{0.3em}
        \normalsize Good day $\Rightarrow$ positive reward\\[0.3em]
        Bad day $\Rightarrow$ negative reward
        \vspace{0.3em}
      \end{block}
    \end{column}
    \begin{column}{0.32\textwidth}
      \begin{block}{\centering Avoid volatile markets}
        \centering\vspace{0.3em}
        \normalsize The variance comes from a \textbf{rolling 60-day covariance matrix} --- if a market has been wild lately, the penalty is bigger
        \vspace{0.3em}
      \end{block}
    \end{column}
  \end{columns}
  \vspace{0.8em}
  \centering\small\textit{In plain English: pick the market with the best return \textbf{for the risk you are taking on}.}
\end{frame}

\begin{frame}{How PPO Trains the Agent}
  \vspace{0.2em}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \textbf{The loop (runs 10 years of history, 50 times)}\\[0.7em]
      \begin{enumerate}\normalsize
        \setlength\itemsep{0.7em}
        \item \textbf{See} the macro state today (10 numbers)
        \item \textbf{Pick} an ETF based on current policy
        \item \textbf{Get} the reward (return minus risk penalty)
        \item After 252 days, \textbf{update} the policy --- then repeat
      \end{enumerate}
    \end{column}
    \begin{column}{0.48\textwidth}
      \textbf{What makes PPO special}\\[0.7em]
      \begin{itemize}\normalsize
        \setlength\itemsep{0.7em}
        \item \alert{Clipped updates} --- the policy can only change a little each round, so it doesn't overcorrect on one noisy day and forget everything
        \item \alert{Entropy bonus} --- forces the agent to keep exploring all 6 markets rather than always picking the same one
        \item \alert{10 random seeds run} --- best one kept (seed 999, Sharpe 0.75)
      \end{itemize}
    \end{column}
  \end{columns}
  \vspace{1em}
  \begin{center}
    \begin{beamercolorbox}[wd=0.85\textwidth,center,sep=0.4em]{block body}
      \normalsize\textit{Think of PPO as a student who reviews a whole year of trades, takes a cautious step to improve, then reviews again --- never overcorrecting on a single bad day.}
    \end{beamercolorbox}
  \end{center}
\end{frame}

\begin{frame}{What the Model Outputs at Trade Time}
  \begin{columns}[T]
    \begin{column}{0.50\textwidth}
      \textbf{Inference pipeline}\\[0.6em]
      \begin{enumerate}\normalsize
        \setlength\itemsep{0.6em}
        \item Observe the 10-feature macro state at close of day $t$
        \item Feed into the Policy Network $\pi_\theta(s_t)$
        \item Softmax output: a probability over all 6 ETFs
        \item Take the \textbf{argmax} --- the highest-probability ETF
        \item Allocate \textbf{100\%} of capital to that ETF
        \item Hold through day $t+1$, then re-evaluate
      \end{enumerate}
      \vspace{0.5em}
      \small\textit{The portfolio is always fully invested in exactly one position. No partial allocations.}
    \end{column}
    \begin{column}{0.46\textwidth}
      \begin{block}{Example: a typical trade signal}
        \vspace{0.3em}\small
        State on a given day:\\[0.2em]
        \quad Fed assets $\uparrow$, M2 $\uparrow$, VIX rolling over,\\
        \quad yield curve steepening, INDA BMOM $\uparrow\uparrow$\\[0.5em]
        Policy output:
        \begin{center}
        \small\begin{tabular}{lc}
          \toprule
          ETF & Probability \\
          \midrule
          SPY  & 0.18 \\
          EWJ  & 0.06 \\
          \rowcolor{blue!10}
          \textbf{INDA} & \textbf{0.51} \\
          MCHI & 0.11 \\
          EZU  & 0.07 \\
          BIL  & 0.07 \\
          \bottomrule
        \end{tabular}
        \end{center}
        $\Rightarrow$ \textbf{Hold INDA tomorrow.}
        \vspace{0.3em}
      \end{block}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Why the Hybrid --- RL Alone Is Not Enough}
  \begin{columns}[T]
    \begin{column}{0.50\textwidth}
      \textbf{The problem with RL alone}\\[0.4em]
      \normalsize
      The RL model was trained on 2004--2013 (GFC and early recovery). If market dynamics shift significantly --- new volatility regimes, unprecedented macro combinations --- the learned policy can \textbf{over-extend drawdowns} before adapting.\\[0.6em]
      \textbf{What the hybrid adds:}
      \begin{itemize}\normalsize
        \setlength\itemsep{0.4em}
        \item A \textbf{drawdown-differential regime switch}: when the RL model's running drawdown exceeds the BMOM baseline's drawdown by $>8\%$, automatically switch to momentum
        \item When the gap closes back to $<3\%$, switch back to RL
        \item BMOM is \textbf{regime-agnostic} --- it always follows current price trends, regardless of what macro regime it is
      \end{itemize}
    \end{column}
    \begin{column}{0.46\textwidth}
      \textbf{The result}\\[0.4em]
      \begin{center}\small
      \begin{tabular}{lcc}
        \toprule
        Strategy & Sharpe & Max DD \\
        \midrule
        RL standalone & 0.75 & $-$34.6\% \\
        \rowcolor{green!8}
        \textbf{Hybrid} & \textbf{0.73} & $\mathbf{-}$\textbf{30.0\%} \\
        BMOM baseline  & 0.45 & $-$28.6\% \\
        \bottomrule
      \end{tabular}
      \end{center}
      \vspace{0.5em}
      \normalsize
      The hybrid is \textbf{in RL mode 95.6\% of the time} --- the switch rarely activates, but when it does (e.g.\ COVID crash, 2022 rate shock) it absorbs the worst of the drawdown.\\[0.5em]
      \small
      Think of it as: \textit{RL handles macro regime identification (better returns), BMOM handles tail risk (lower drawdowns).}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Walk-Forward Validation --- No Peeking}
  \begin{columns}[T]
    \begin{column}{0.52\textwidth}
      \textbf{The most important discipline:}\\[0.6em]\normalsize
      The model was trained \textbf{only on 2004--2013}.\\[0.5em]
      Everything from \textbf{2014 to 2025} --- 11 years --- is out-of-sample. The model had \textbf{never seen} this data.\\[0.5em]
      This means the results are real, not overfitting.\\[0.8em]
      \textbf{The OOS period covers:}
      \begin{itemize}\small
        \setlength\itemsep{0.2em}
        \item Post-GFC bull market (2014--2019)
        \item COVID crash and recovery (2020--2021)
        \item Rate hike cycle (2022)
        \item AI-driven rally (2023--2025)
      \end{itemize}
    \end{column}
    \begin{column}{0.44\textwidth}
      \begin{block}{\centering Training split}
